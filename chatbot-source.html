Chatbot IA — Portfolio Lucas Fouquet

Chatbot intégré au portfolio, alimenté par l'API Mistral AI. Il répond en première personne au nom de Lucas Fouquet et se limite aux questions relatives à son parcours et ses projets. Implémenté en vanilla JavaScript, sans framework ni dépendance externe.

Stack technique
Modèle LLM
open-mistral-nemo — Mistral AI (open-source, gratuit)
API
Mistral AI REST API — appel direct côté navigateur
Approche RAG
Knowledge base statique injectée dans le system prompt (7 projets documentés). Adapté à une base de connaissance fixe et de taille réduite.
Mémoire
Historique conversationnel maintenu côté client (tableau history[]) — le contexte est renvoyé à chaque appel
Quota
10 messages maximum par session — protection contre les coûts excessifs
Dépendances
Zéro — vanilla JavaScript pur, aucun framework

Architecture du system prompt

Le chatbot parle à la première personne. Toute la connaissance (projets, compétences, parcours) est injectée dans le system prompt à chaque requête. Les questions hors-sujet reçoivent une réponse humoristique qui redirige vers le portfolio.

// ═══════════════════════════════════════════════════════════
// CHATBOT LUCAS FOUQUET — Mistral AI
// Widget flottant vanilla JS, aucune dépendance externe
// ═══════════════════════════════════════════════════════════

(function() {

  // ── Configuration ─────────────────────────────────────────
  var MISTRAL_KEY  = 'VOTRE_CLÉ_API_MISTRAL';
  var MAX_MESSAGES = 10;
  var msgCount     = 0;

  // ── System prompt (knowledge base) ────────────────────────
  // Le LLM parle à la 1ère personne au nom de Lucas.
  // Toute la connaissance est injectée ici statiquement.
  var SYSTEM_PROMPT = `Tu es Lucas Fouquet. Tu parles en ton propre nom, à la première
personne. Un visiteur de ton portfolio te pose des questions. Réponds
comme si c'était toi — naturel, direct, concis. Réponds en français
sauf si on te parle en anglais.

Si la question est hors-sujet : réponds avec humour et bienveillance
que tu es là pour parler de ton parcours et tes projets.

TON PROFIL :
- Étudiant en Data & IA à Eugenia School (Paris)
- Compétences : Python, SQL, Tableau, Excel, LangChain, RAG,
  Make (no-code), vibe coding avec Claude
- Email : lfouquet@eugeniaschool.com
- Projet préféré : Payfit SEO (no-code, déployé en conditions réelles)

PROJETS :
1. Discogs Scraper  — Python, scraping 10 000 albums Discogs
2. Synapse          — RAG hybride FAISS+BM25, analyse de thèses HAL
3. Accidents France — SQL, analyse BAAC 2020-2024, 611 762 usagers
4. Olist            — Tableau Public, e-commerce Brésil 2016-2018
5. Paris Sportifs   — Excel, Top 5 ligues européennes
6. Together         — Next.js 14, Supabase, app mobile sociale
7. Payfit SEO       — Make + Dust + GPT-4o-mini, veille SEO automatisée

[...system prompt complet dans index.html]`;

  // ── Références DOM ────────────────────────────────────────
  var fab      = document.getElementById('chat-fab');
  var panel    = document.getElementById('chat-panel');
  var closeBtn = document.getElementById('chat-close');
  var messages = document.getElementById('chat-messages');
  var input    = document.getElementById('chat-input');
  var sendBtn  = document.getElementById('chat-send');
  var quota    = document.getElementById('chat-quota');

  // Historique pour le contexte multi-tours
  var history = [];

  // ── Toggle panel ──────────────────────────────────────────
  fab.addEventListener('click', function() {
    panel.classList.toggle('open');
    if (panel.classList.contains('open')) input.focus();
  });
  closeBtn.addEventListener('click', function() {
    panel.classList.remove('open');
  });

  // ── Envoi d'un message ────────────────────────────────────
  function sendMessage() {
    var text = input.value.trim();
    if (!text || msgCount >= MAX_MESSAGES) return;

    appendMsg('user', text);
    history.push({ role: 'user', content: text });
    input.value = '';
    sendBtn.disabled = true;

    var typingEl = appendTyping();

    // Appel API Mistral
    fetch('https://api.mistral.ai/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer ' + MISTRAL_KEY
      },
      body: JSON.stringify({
        model:       'open-mistral-nemo',
        messages:    [{ role: 'system', content: SYSTEM_PROMPT }].concat(history),
        max_tokens:  400,
        temperature: 0.4
      })
    })
    .then(function(r) {
      return r.json().then(function(data) {
        if (!r.ok) { console.error('[Chat] API', r.status, data); throw data; }
        return data;
      });
    })
    .then(function(data) {
      typingEl.remove();
      var reply = (data.choices && data.choices[0] &&
                   data.choices[0].message &&
                   data.choices[0].message.content)
        ? data.choices[0].message.content
        : 'Hmm, réponse inattendue. Réessaie !';
      appendMsg('bot', reply);
      history.push({ role: 'assistant', content: reply });
      msgCount++;
      updateQuota();
      sendBtn.disabled = (msgCount >= MAX_MESSAGES);
    })
    .catch(function(err) {
      typingEl.remove();
      console.error('[Chat] Network error:', err);
      appendMsg('bot',
        'Oups, une erreur réseau — réessaie dans quelques secondes. ' +
        'Si ça persiste : lfouquet@eugeniaschool.com'
      );
      sendBtn.disabled = false;
    });
  }

  sendBtn.addEventListener('click', sendMessage);
  input.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') sendMessage();
  });

  // ── Helpers ───────────────────────────────────────────────
  function appendMsg(role, text) {
    var d = document.createElement('div');
    d.className = 'cm ' + role;
    var b = document.createElement('div');
    b.className = 'cm-bubble';
    b.textContent = text;
    d.appendChild(b);
    messages.appendChild(d);
    messages.scrollTop = messages.scrollHeight;
    return d;
  }

  function appendTyping() {
    var d = document.createElement('div');
    d.className = 'cm bot';
    d.innerHTML = '<div class="cm-bubble">' +
      '<div class="typing-dots">' +
      '<span></span><span></span><span></span>' +
      '</div></div>';
    messages.appendChild(d);
    messages.scrollTop = messages.scrollHeight;
    return d;
  }

  function updateQuota() {
    var left = MAX_MESSAGES - msgCount;
    quota.textContent = left > 0
      ? left + ' message' + (left > 1 ? 's' : '') +
        ' restant' + (left > 1 ? 's' : '')
      : 'Quota atteint — rechargez la page pour recommencer';
  }

})();
